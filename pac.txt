const domainObject = {};
domains.filter(Boolean).forEach(domain => {
	domainObject[domain] = 1;
});

const proxy = "SOCKS5 127.0.0.1:1080;";
const direct = 'DIRECT;';
function FindProxyForURL(url, host) {
	let suffix;
	let pos = host.lastIndexOf('.');
	pos = host.lastIndexOf('.', pos - 1);
	while(1) {
		if (pos <= 0) {
			if (Object.hasOwnProperty.call(domainObject, host)) { return proxy; } 
			else { return direct; }
		}
		suffix = host.substring(pos + 1);
		if (Object.hasOwnProperty.call(domainObject, suffix)) { return proxy; }
		pos = host.lastIndexOf('.', pos - 1);
	}
}
